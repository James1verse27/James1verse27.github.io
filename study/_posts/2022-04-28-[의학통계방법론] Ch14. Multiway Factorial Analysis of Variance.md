oxygen <- ex14_1a$Y

anova(aov(oxygen ~ species*temperature*sex)) %>% 
  tidy() %>% 
  rename(" "="term","Sum Sq"="sumsq","Mean Sq"="meansq","F value"="statistic","Pr(>F)"="p.value") %>%
